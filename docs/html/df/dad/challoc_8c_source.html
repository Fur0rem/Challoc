108ClosePowerOfTwo is_close_to_power_of_two(size_t size) {
109 ClosePowerOfTwo false_result = {
110 .is_close = false,
111 .ceil_pow2 = 0,
112 };
113
114 // Will not fit in the minislab
115 if (size > 512) {
116 return false_result;
117 }
118
119 // Accept to allocate in the minislab to any size smaller than 4
120 if (size <= 4) {
121 return (ClosePowerOfTwo){
122 .is_close = true,
123 .ceil_pow2 = 2,
124 };
125 }
126
127 // Compute the smallest power of two that is greater than the size
128 size_t ceil_pow2 = 2 * 2;
129 size_t pow2 = 2;
130 while (ceil_pow2 < size) {
131 ceil_pow2 *= 2;
132 pow2++;
133 }
134
135 // If the size is close to the power of two, we accept it
136 const double threshold = 1.2;
137 if ((double)ceil_pow2 / (double)size <= threshold) {
138 return (ClosePowerOfTwo){
139 .is_close = true,
140 .ceil_pow2 = pow2,
141 };
142 }
143 else {
144 return false_result;
145 }
146}

268size_t minislab_ptr_size(void* ptr) {

425size_t time_to_live_with_size(size_t size) {
426 size_t ceil = ceil_to_4096multiple(size);
427 size_t log2 = 4096; // size of a page
428 size_t ttl = 5; // 5 is the max time to live
429
430 // Compute the log2 of the size
431 while (log2 < ceil) {
432 log2 *= 2;
433 ttl--;
434 if (ttl == 1) {
435 break;
436 }
437 }
438
439 return ttl;

593void* try_allocate_next_to(AllocMetadata* metadata, size_t size_requested, size_t size_needed, size_t block_idx) {
594 Block* block = &challoc_blocks_in_use.blocks[block_idx];
595 AllocMetadata* next = metadata->next;
596 if (next == NULL) { // We are at the end of the linked list
597 size_t space_between_last_and_end_of_block =
598 ((size_t)block->mmap_ptr + block->size) - (size_t)metadata - sizeof(AllocMetadata) - metadata->size;
599 if (space_between_last_and_end_of_block >= size_needed) {
600 AllocMetadata* new_metadata = (AllocMetadata*)((size_t)metadata + sizeof(AllocMetadata) + metadata->size);
601 new_metadata->size = size_requested;
602 new_metadata->next = NULL;
603 new_metadata->prev = metadata;
604 new_metadata->block_idx = block_idx;
605
606 metadata->next = new_metadata;
607 block->tail = new_metadata;
608 block->free_space -= size_needed;
609 assert(block->free_space <= block->size);
610 block->freshly_allocated = false;
611 return (uint8_t*)(new_metadata) + sizeof(AllocMetadata);
612 }
613 }
614 else { // We are in the middle of the linked list
615 ssize_t space_between_current_and_next = (ssize_t)next - (ssize_t)metadata - sizeof(AllocMetadata) - metadata->size;
616 if (space_between_current_and_next >= (ssize_t)size_needed) { // We can allocate in between
617 AllocMetadata* new_metadata = (AllocMetadata*)((size_t)metadata + sizeof(AllocMetadata) + metadata->size);
618 new_metadata->size = size_requested;
619 new_metadata->next = next;
620 new_metadata->prev = metadata;
621 new_metadata->block_idx = block_idx;
622
623 metadata->next = new_metadata;
624 next->prev = new_metadata;
625 block->free_space -= size_needed;
626 assert(block->free_space <= block->size);
627 block->freshly_allocated = false;
628 return (uint8_t*)(new_metadata) + sizeof(AllocMetadata);
629 }
630 }
631 return NULL;
632}

641void* block_try_allocate(BlockList* list, size_t block_idx, size_t size_requested) {
642 assert(block_idx <= list->size);
643 assert(list->blocks[block_idx].free_space <= list->blocks[block_idx].size);
644 size_t size_needed = size_requested + sizeof(AllocMetadata); // We need to store the metadata
645
646 // We will never be able to find a space
647 Block* block = &list->blocks[block_idx];
648 if (block->free_space < size_needed) {
649 return NULL;
650 }
651
652 // First allocated block
653 AllocMetadata* last = block->tail;
654 if (last == NULL) {
655 block->head = (AllocMetadata*)(block->mmap_ptr);
656 block->head->size = size_requested;
657 block->head->next = NULL;
658 block->head->prev = NULL;
659 block->head->block_idx = block_idx;
660 block->tail = block->head;
661 block->free_space -= size_needed;
662 assert(block->free_space <= block->size);
663 block->freshly_allocated = false;
664 return (uint8_t*)(block->head) + sizeof(AllocMetadata);
665 }
666
667 // Go through the linked list to find a space
668 for (AllocMetadata* current = block->head; current != NULL; current = current->next) {
669 void* ptr = try_allocate_next_to(current, size_requested, size_needed, block_idx);
670 if (ptr != NULL) {
671 return ptr;
672 }
673 }
674
675 // We did not find a space
676 return NULL;
677}

684void block_free(Block* block, AllocMetadata* ptr) {
685 AllocMetadata* current = block->head;
686
687 // Check if the only block is the one we want to free
688 if (current == ptr) {
689 block->head = current->next;
690 if (block->head == NULL) {
691 block->tail = NULL;
692 }
693 block->free_space += current->size + sizeof(AllocMetadata);
694 assert(block->free_space <= block->size);
695 return;
696 }
697
698 // Remove the metadata from the linked list
699 if (ptr->prev != NULL) {
700 ptr->prev->next = ptr->next;
701 }
702 if (ptr->next != NULL) {
703 ptr->next->prev = ptr->prev;
704 }
705
706 // Update the head and tail
707 if (ptr == block->head) {
708 block->head = ptr->next;
709 }
710
711 if (ptr == block->tail) {
712 block->tail = ptr->prev;
713 }
714
715 // Update the free space
716 block->free_space += ptr->size + sizeof(AllocMetadata);
717 assert(block->free_space <= block->size);
718}

849 challoc_freed_blocks.size--;

851 // Revive the block into a ready-to-use one
852 block.time_to_live = time_to_live_with_size(block.size);
853 block.head = NULL;
854 block.tail = NULL;
855 block.free_space = block.size;
856 blocklist_push(&challoc_blocks_in_use, block);
857 void* ptr = block_try_allocate(&challoc_blocks_in_use, challoc_blocks_in_use.size - 1, size);
858 decrease_ttl_and_unmap();
859 last_block_alloc = ptr;
860 return ptr;
861 }
862 }
863
864 // No block had enough space, create a new one
865 blocklist_allocate_new_block(&challoc_blocks_in_use, size + sizeof(AllocMetadata) + sizeof(Block));
866
867 // Check if we could allocate the new one
868 if (challoc_blocks_in_use.blocks == MAP_FAILED) {
869 return NULL;
870 }
871
872 void* ptr = block_try_allocate(&challoc_blocks_in_use, challoc_blocks_in_use.size - 1, size);
873
874 decrease_ttl_and_unmap();
875
876 last_block_alloc = ptr;
877 return ptr;
878}

884void __chafree(void* ptr) {
885 if (ptr == NULL) {
886 return;
887 }
888
889 // Check if the pointer comes from the minislab before getting wrong metadata
890 if (ptr_comes_from_minislab(ptr)) {
891 minislab_free(ptr);
892 return;
893 }
894
895 if (last_block_alloc == ptr) {
896 last_block_alloc = NULL;
897 }
898
899 AllocMetadata* metadata = challoc_get_metadata(ptr);
900
901 if (metadata->prev != NULL) {
902 prev_of_last_block_free = metadata->prev;
903 }
904
905 // Free the memory from the block list
906 blocklist_free(metadata);
907}

914void* __chacalloc(size_t nmemb, size_t size) {
915 // Allocate the memory
916 void* ptr = __chamalloc(nmemb * size);
917
918 // Check if the pointer comes from the minislab before getting wrong metadata
919 if (ptr_comes_from_minislab(ptr)) {
920 memset(ptr, 0, nmemb * size);
921 return ptr;
922 }
923
924 // Check if it comes from a freshly allocated block
925 AllocMetadata* metadata = challoc_get_metadata(ptr);
926 size_t block_idx = metadata->block_idx;
927 Block* block = &challoc_blocks_in_use.blocks[block_idx];
928 if (block->freshly_allocated) { // No need to memset, the block is already zeroed
929 for (size_t i = 0; i < nmemb * size; i++) {
930 assert(((uint8_t*)ptr)[i] == 0);
931 }
932 return ptr;
933 }
934
935 // Set memory to zero
936 if (ptr != NULL) {
937 memset(ptr, 0, nmemb * size);
938 }
939
940 return ptr;
941}

949void* __charealloc(void* ptr, size_t new_size) {
950 // If ptr is NULL, behave like malloc
951 if (ptr == NULL) {
952 return __chamalloc(new_size);
953 }
954
955 // Get the size of the allocation
956 size_t old_size = 0;
957 if (ptr_comes_from_minislab(ptr)) {
958 old_size = minislab_ptr_size(ptr);
959 }
960 else {
961 AllocMetadata* metadata = challoc_get_metadata(ptr);
962 old_size = metadata->size;
963 }
964
965 void* new_ptr = __chamalloc(new_size);
966 if (new_ptr == NULL) {
967 return NULL;
968 }
969
970 // Copy the old data
971 size_t min_size = old_size < new_size ? old_size : new_size;
972 memcpy(new_ptr, ptr, min_size);
973
974 // Free the old memory
975 __chafree(ptr);
976
977 return new_ptr;
978}
