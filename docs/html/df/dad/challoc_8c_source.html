108ClosePowerOfTwo is_close_to_power_of_two(size_t size) {
109 ClosePowerOfTwo false_result = {
110 .is_close = false,
111 .ceil_pow2 = 0,
112 };
113
114 // Will not fit in the minislab
115 if (size > 512) {
116 return false_result;
117 }
118
119 // Accept to allocate in the minislab to any size smaller than 4
120 if (size <= 4) {
121 return (ClosePowerOfTwo){
122 .is_close = true,
123 .ceil_pow2 = 2,
124 };
125 }
126
127 // Compute the smallest power of two that is greater than the size
128 size_t ceil_pow2 = 2 * 2;
129 size_t pow2 = 2;
130 while (ceil_pow2 < size) {
131 ceil_pow2 *= 2;
132 pow2++;
133 }
134
135 // If the size is close to the power of two, we accept it
136 const double threshold = 1.2;
137 if ((double)ceil_pow2 / (double)size <= threshold) {
138 return (ClosePowerOfTwo){
139 .is_close = true,
140 .ceil_pow2 = pow2,
141 };
142 }
143 else {
144 return false_result;
145 }
146}

170void* minislab_alloc(ClosePowerOfTwo size) {
171 // If we are here, we have accepted to allocate in the minislab
172 // And therefore the size is between 4 and 512
173 assert(size.is_close);
174 assert(size.ceil_pow2 >= 2 && size.ceil_pow2 <= 9);

177 case 2: { // 4 bytes

268size_t minislab_ptr_size(void* ptr) {
269 assert(ptr_comes_from_minislab(ptr));
270 ssize_t offset = (ssize_t)ptr - (ssize_t)&challoc_minislab;
271 assert(offset < 4096);
272
273 static const int sizes[] = {512, 256, 128, 64, 32, 16, 8, 4}; // Sizes of the layers
274 static const int layer_offsets[] = {512, 1024, 1536, 2048, 2560, 3072, 3584, 4096}; // Offsets of the layers
275
276 // Find the layer of the pointer
277 if (offset >= 0 && offset < layer_offsets[0]) {
278 return 512;
279 }
280 for (int i = 1; i < 8; i++) {
281 if (offset >= layer_offsets[i - 1] && offset < layer_offsets[i]) {
282 return sizes[i];
283 }
284 }
285
286 return SIZE_MAX; // Should not reach here

422size_t time_to_live_with_size(size_t size) {
423 size_t ceil = ceil_to_4096multiple(size);
424 size_t log2 = 4096; // size of a page
425 size_t ttl = 5; // 5 is the max time to live
426
427 // Compute the log2 of the size
428 while (log2 < ceil) {
429 log2 *= 2;
430 ttl--;
431 if (ttl == 1) {
432 break;
433 }
434 }
435
436 return ttl;
437}

590void* block_try_allocate(BlockList* list, size_t block_idx, size_t size_requested) {
591 assert(block_idx <= list->size);
592 assert(list->blocks[block_idx].free_space <= list->blocks[block_idx].size);
593 size_t size_needed = size_requested + sizeof(AllocMetadata); // We need to store the metadata
594
595 // We will never be able to find a space
596 Block* block = &list->blocks[block_idx];
597 if (block->free_space < size_needed) {
598 return NULL;
599 }
600
601 // First allocated block
602 AllocMetadata* last = block->tail;
603 if (last == NULL) {
604 block->head = (AllocMetadata*)(block->mmap_ptr);
605 block->head->size = size_requested;
606 block->head->next = NULL;
607 block->head->prev = NULL;
608 block->head->block_idx = block_idx;
609 block->tail = block->head;
610 block->free_space -= size_needed;
611 assert(block->free_space <= block->size);
612 block->freshly_allocated = false;
613 return (uint8_t*)(block->head) + sizeof(AllocMetadata);
614 }
615
616 // Go through the linked list to find a space
617 for (AllocMetadata* current = block->head; current != NULL; current = current->next) {
618 AllocMetadata* next = current->next;
619 if (next == NULL) { // We are at the end of the linked list
620 size_t space_between_last_and_end_of_block =
621 ((size_t)block->mmap_ptr + block->size) - (size_t)current - sizeof(AllocMetadata) - current->size;
622 if (space_between_last_and_end_of_block >= size_needed) {
623 AllocMetadata* new_metadata = (AllocMetadata*)((size_t)current + sizeof(AllocMetadata) + current->size);
624 new_metadata->size = size_requested;
625 new_metadata->next = NULL;
626 new_metadata->prev = current;
627 new_metadata->block_idx = block_idx;
628
629 current->next = new_metadata;
630 block->tail = new_metadata;
631 block->free_space -= size_needed;
632 assert(block->free_space <= block->size);
633 block->freshly_allocated = false;
634 return (uint8_t*)(new_metadata) + sizeof(AllocMetadata);
635 }
636 }
637 else { // We are in the middle of the linked list
638 ssize_t space_between_current_and_next = (ssize_t)next - (ssize_t)current - sizeof(AllocMetadata) - current->size;
639 if (space_between_current_and_next >= (ssize_t)size_needed) { // We can allocate in between
640 AllocMetadata* new_metadata = (AllocMetadata*)((size_t)current + sizeof(AllocMetadata) + current->size);
641 new_metadata->size = size_requested;
642 new_metadata->next = next;
643 new_metadata->prev = current;
644 new_metadata->block_idx = block_idx;
645
646 current->next = new_metadata;
647 next->prev = new_metadata;
648 block->free_space -= size_needed;
649 assert(block->free_space <= block->size);
650 block->freshly_allocated = false;
651 return (uint8_t*)(new_metadata) + sizeof(AllocMetadata);
652 }
653 }
654 }
655
656 // We did not find a space
657 return NULL;
658}

665void block_free(Block* block, AllocMetadata* ptr) {
666 AllocMetadata* current = block->head;
667
668 // Check if the only block is the one we want to free
669 if (current == ptr) {
670 block->head = current->next;
671 if (block->head == NULL) {
672 block->tail = NULL;
673 }
674 block->free_space += current->size + sizeof(AllocMetadata);
675 assert(block->free_space <= block->size);
676 return;
677 }
678
679 // Remove the metadata from the linked list
680 if (ptr->prev != NULL) {
681 ptr->prev->next = ptr->next;
682 }
683 if (ptr->next != NULL) {
684 ptr->next->prev = ptr->prev;
685 }
686
687 // Update the head and tail
688 if (ptr == block->head) {
689 block->head = ptr->next;
690 }
691
692 if (ptr == block->tail) {
693 block->tail = ptr->prev;
694 }
695
696 // Update the free space
697 block->free_space += ptr->size + sizeof(AllocMetadata);
698 assert(block->free_space <= block->size);

708 Block* block = &challoc_blocks_in_use.blocks[block_idx];
709 block_free(block, ptr);

728 challoc_blocks_in_use.size--;

772void* __chamalloc(size_t size) {
773 if (size == 0) {
774 return NULL;
775 }
776
777 // Try to allocate from the minislab
778 ClosePowerOfTwo close_pow2 = is_close_to_power_of_two(size);
779 if (close_pow2.is_close) {
780 void* ptr = minislab_alloc(close_pow2);
781 if (ptr != NULL) {
782 decrease_ttl_and_unmap();
783 return ptr;
784 }
785 }
786
787 // Go through the list of mmap blocks
788 for (size_t i = 0; i < challoc_blocks_in_use.size; i++) {
789 Block* block = &challoc_blocks_in_use.blocks[i];
790 if (block_has_enough_space(block, size)) {
791 void* ptr = block_try_allocate(&challoc_blocks_in_use, i, size);
792 if (ptr != NULL) {
793 decrease_ttl_and_unmap();
794 return ptr;
795 }
796 }
797 }
798
799 // Go through the list of freed blocks
800 for (size_t i = 0; i < challoc_freed_blocks.size; i++) {
801 Block block = challoc_freed_blocks.blocks[i];
802 if (block_has_enough_space(&block, size)) {
803 Block* last_block = &challoc_freed_blocks.blocks[challoc_freed_blocks.size - 1];
804 challoc_freed_blocks.blocks[i] = *last_block;
805 challoc_freed_blocks.size--;
806
807 // Revive the block into a ready-to-use one
808 block.time_to_live = time_to_live_with_size(block.size);
809 block.head = NULL;
810 block.tail = NULL;
811 block.free_space = block.size;
812 blocklist_push(&challoc_blocks_in_use, block);
813 void* ptr = block_try_allocate(&challoc_blocks_in_use, challoc_blocks_in_use.size - 1, size);
814 decrease_ttl_and_unmap();
815 return ptr;
816 }
817 }
818
819 // No block had enough space, create a new one
820 blocklist_allocate_new_block(&challoc_blocks_in_use, size + sizeof(AllocMetadata) + sizeof(Block));
821
822 // Check if we could allocate the new one
823 if (challoc_blocks_in_use.blocks == MAP_FAILED) {
824 return NULL;
825 }
826
827 void* ptr = block_try_allocate(&challoc_blocks_in_use, challoc_blocks_in_use.size - 1, size);
828
829 decrease_ttl_and_unmap();
830
831 return ptr;
832}

859void* __chacalloc(size_t nmemb, size_t size) {
860 // Allocate the memory
861 void* ptr = __chamalloc(nmemb * size);
862
863 // Check if the pointer comes from the minislab before getting wrong metadata
864 if (ptr_comes_from_minislab(ptr)) {
865 memset(ptr, 0, nmemb * size);
866 return ptr;
867 }
868
869 // Check if it comes from a freshly allocated block
870 AllocMetadata* metadata = challoc_get_metadata(ptr);
871 size_t block_idx = metadata->block_idx;
872 Block* block = &challoc_blocks_in_use.blocks[block_idx];
873 if (block->freshly_allocated) { // No need to memset, the block is already zeroed
874 for (size_t i = 0; i < nmemb * size; i++) {
875 assert(((uint8_t*)ptr)[i] == 0);
876 }
877 return ptr;
878 }
879
880 // Set memory to zero
881 if (ptr != NULL) {
882 memset(ptr, 0, nmemb * size);
883 }
884
885 return ptr;
886}

894void* __charealloc(void* ptr, size_t new_size) {
895 // If ptr is NULL, behave like malloc
896 if (ptr == NULL) {
897 return __chamalloc(new_size);
898 }
899
900 // Get the size of the allocation
901 size_t old_size = 0;
902 if (ptr_comes_from_minislab(ptr)) {
903 old_size = minislab_ptr_size(ptr);
904 }
905 else {
906 AllocMetadata* metadata = challoc_get_metadata(ptr);
907 old_size = metadata->size;
908 }
909
910 void* new_ptr = __chamalloc(new_size);
911 if (new_ptr == NULL) {
912 return NULL;
913 }
914
915 // Copy the old data
916 size_t min_size = old_size < new_size ? old_size : new_size;
917 memcpy(new_ptr, ptr, min_size);
918
919 // Free the old memory
920 __chafree(ptr);
921
922 return new_ptr;
923}
